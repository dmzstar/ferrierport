我们来测试一下Vue2的生命周期方法，下面是官方文档展示的生命周期流程图

完整的生命周期流程例子

        
		var vm = new Vue({
			el : '.mainBody',
			template : "{{name}}",
			data : {
				'name' : 'Vue Lifecycle show'
			},
			beforeCreate : function() {
				console.log('beforeCreated:', arguments);
			},
			created : function() {
				console.log('created:', arguments);
			},
			beforeMount : function() {
				console.log('beforeMount:', arguments);
				console.log('beforeMount方法只有在设置el属性的时候才会调用。');
			},
			mounted : function() {
				console.log('mounted:', arguments);
				console.log('mounted在mount阶段完成之后调用。');
			},
			beforeUpdate : function() {
				console.log('beforeUpdate:', arguments);
				console.log('beforeUpdate钩子方法只有在设置了template并且属性值有更改时才会调用。');
			},
			updated : function() {
				console.log('updated:', arguments);
				console.log('updated在updated阶段完成后调用。');
			}
		});
